Step 2: Signal Processing in Python (on collected data)
python
CopyEdit
import numpy as np
from scipy.signal import butter, filtfilt

# Simulated IR values (from sensor)
raw_signal = np.loadtxt("ppg_ir_signal.csv")

# Apply Butterworth filter
def butter_lowpass_filter(data, cutoff=2.5, fs=25.0, order=4):
    nyq = 0.5 * fs
    normal_cutoff = cutoff / nyq
    b, a = butter(order, normal_cutoff, btype='low', analog=False)
    return filtfilt(b, a, data)

filtered = butter_lowpass_filter(raw_signal)

# Now use peak detection for heart rate
from scipy.signal import find_peaks

peaks, _ = find_peaks(filtered, distance=25)
heart_rate = (len(peaks) / (len(raw_signal)/25)) * 60  # BPM
print(f"Heart Rate: {heart_rate:.2f} bpm")

Step 7: Flutter App to Receive BLE Alerts

// Flutter using flutter_blue_plus package
FlutterBluePlus.instance.scanResults.listen((results) {
  for (ScanResult r in results) {
    if (r.device.name == "WearableMonitor") {
      r.device.connect();
      // Subscribe to characteristic
      r.device.discoverServices().then((services) {
        final characteristic = services.first.characteristics.first;
        characteristic.setNotifyValue(true);
        characteristic.value.listen((value) {
          final message = utf8.decode(value);
          print("ALERT: $message");
        });
      });
    }
  }
});

import board
import adafruit_dht

while True:
    try:
        temp_c = dhtDevice.temperature
        humidity = dhtDevice.humidity
        payload = f"{temp_c},{humidity}"
        client.publish("machine/data", payload)
    except RuntimeError:
        continue
    time.sleep(2)

import paho.mqtt.client as mqtt
import json
import requests

def on_message(client, userdata, msg):
    data = msg.payload.decode()
    vibration, temp, hum = map(float, data.split(","))
    alert = {
        "vibration": vibration,
        "temperature": temp,
        "humidity": hum,
        "status": "ok" if vibration < 0.8 else "alert"
    }
    requests.post("https://your-cloud-api.com/ingest", json=alert)

client = mqtt.Client()

client.subscribe("machine/data")
client.on_message = on_message
client.loop_forever()

import smbus
import time

bus = smbus.SMBus(1)
address = 0x53  # Confirm with i2cdetect

# Power on
bus.write_byte_data(address, 0x2D, 0x08)

def read_axis(axis):
    reg = {'x': 0x32, 'y': 0x34, 'z': 0x36}[axis]
    low = bus.read_byte_data(address, reg)
    high = bus.read_byte_data(address, reg+1)
    val = ((high << 8) | low)
    return val - 65536 if val > 32767 else val

while True:
    print("X:", read_axis('x'), "Y:", read_axis('y'), "Z:", read_axis('z'))
    time.sleep(0.5)

python
CopyEdit
import smbus2
import time

bus = smbus2.SMBus(1)
address = 0x53

# Initialize ADXL345
bus.write_byte_data(address, 0x2D, 0x08)  # Power control register to start measurements

def read_word(reg):
    low = bus.read_byte_data(address, reg)
    high = bus.read_byte_data(address, reg+1)
    val = (high << 8) + low
    return val - 65536 if val > 32767 else val

while True:
    x = read_word(0x32)
    y = read_word(0x34)
    z = read_word(0x36)
    print(f"X: {x}, Y: {y}, Z: {z}")
    time.sleep(0.5)
This reads raw acceleration on X, Y, Z axes every 0.5 seconds.

Feature Extraction & Data Logger — Making Sense of Raw Data
Goal: Convert raw sensor readings into meaningful features, save data for later model training.
Step 1: Collect sensor data and extract features like mean, std, kurtosis
python
CopyEdit
import numpy as np
import time
from collections import deque

window_size = 50  # Number of samples to consider for feature extraction
data_x, data_y, data_z = deque(maxlen=window_size), deque(maxlen=window_size), deque(maxlen=window_size)

def extract_features(window):
    arr = np.array(window)
    features = {
        "mean": np.mean(arr),
        "std": np.std(arr),
        "max": np.max(arr),
        "min": np.min(arr),
        "kurtosis": np.mean((arr - np.mean(arr))**4) / (np.std(arr)**4)
    }
    return features

while True:
    # Collect raw data (use the sensor reading function from previous phase)
    x, y, z = read_word(0x32), read_word(0x34), read_word(0x36)
    data_x.append(x)
    data_y.append(y)
    data_z.append(z)

    if len(data_x) == window_size:
        features_x = extract_features(data_x)
        features_y = extract_features(data_y)
        features_z = extract_features(data_z)

        # Combine features into a single vector
        feature_vector = {
            'x_mean': features_x['mean'],
            'x_std': features_x['std'],
            'x_max': features_x['max'],
            'x_min': features_x['min'],
            'x_kurtosis': features_x['kurtosis'],
            'y_mean': features_y['mean'],
            'y_std': features_y['std'],
            'y_max': features_y['max'],
            'y_min': features_y['min'],
            'y_kurtosis': features_y['kurtosis'],
            'z_mean': features_z['mean'],
            'z_std': features_z['std'],
            'z_max': features_z['max'],
            'z_min': features_z['min'],
            'z_kurtosis': features_z['kurtosis'],
        }

        print(feature_vector)

        # Log to CSV file
        with open("sensor_features.csv", "a") as f:
            f.write(",".join(str(v) for v in feature_vector.values()) + "\n")

    time.sleep(0.1)
Run this and collect data over different operating conditions (normal vs. faulty).

ML Model Training — Teaching the Machine to Predict Failures

CopyEdit
import pandas as pd
from sklearn.ensemble import IsolationForest
from sklearn.model_selection import train_test_split
import joblib

# Load data
df = pd.read_csv("sensor_features.csv", header=None)
X = df.values

# Train/test split (if labeled, separate normal vs anomaly data)
X_train, X_test = train_test_split(X, test_size=0.2, random_state=42)

# Train model
model = IsolationForest(n_estimators=100, contamination=0.05, random_state=42)
model.fit(X_train)

# Save model
joblib.dump(model, "anomaly_model.pkl")

# Optional: Evaluate
y_pred = model.predict(X_test)
print("Anomalies detected:", sum(y_pred == -1))

import cv2
import torch

# Load the YOLOv8 model (assuming you've trained/converted it)
model = torch.hub.load('ultralytics/yolov5', 'yolov5s', pretrained=True)

# Define class mappings
VEHICLE_CLASSES = ['car', 'truck', 'bus', 'motorcycle']
PEDESTRIAN_CLASS = 'person'

def detect_objects(frame):
    results = model(frame)
    detections = results.pandas().xyxy[0]  # Bounding boxes
    output = {'vehicles': [], 'pedestrians': []}

    for _, row in detections.iterrows():
        label = row['name']
        conf = float(row['confidence'])
        box = [int(row['xmin']), int(row['ymin']), int(row['xmax']), int(row['ymax'])]

        if label in VEHICLE_CLASSES:
            output['vehicles'].append({'label': label, 'box': box, 'confidence': conf})
        elif label == PEDESTRIAN_CLASS:
            output['pedestrians'].append({'box': box, 'confidence': conf})

    return output

import time

def control_traffic_lights(detections):
    vehicle_count = len(detections['vehicles'])
    pedestrian_count = len(detections['pedestrians'])

    if vehicle_count > 10:
        green_time = 45
    elif vehicle_count > 5:
        green_time = 30
    else:
        green_time = 15

    if pedestrian_count > 0:
        green_time += 10  # Extend time for safety

    return green_time

main.py — Orchestrator
python
CopyEdit
import cv2
from camera_feed import get_camera_stream
from inference.object_detection import detect_objects
from traffic_logic.signal_controller import control_traffic_lights
from communication.mqtt_client import send_alert

cap = get_camera_stream(0)

while True:
    ret, frame = cap.read()
    if not ret:
        break

    detections = detect_objects(frame)
    signal_time = control_traffic_lights(detections)

    if len(detections['vehicles']) > 20:
        send_alert({'event': 'Heavy Traffic', 'count': len(detections['vehicles'])})

    print(f"Signal Time: {signal_time}s | Vehicles: {len(detections['vehicles'])}")

    cv2.imshow("Traffic Feed", frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

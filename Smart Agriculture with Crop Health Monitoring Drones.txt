from dronekit import connect, VehicleMode, LocationGlobalRelative
import time

# Connect to the vehicle
vehicle = connect('udp:127.0.0.1:14550', wait_ready=True)

def arm_and_takeoff(target_altitude):
    while not vehicle.is_armable:
        print("Waiting for vehicle to be armable...")
        time.sleep(1)
    vehicle.mode = VehicleMode("GUIDED")
    vehicle.armed = True
    while not vehicle.armed:
        print("Waiting to arm...")
        time.sleep(1)
    vehicle.simple_takeoff(target_altitude)
    while True:
        alt = vehicle.location.global_relative_frame.alt
        print(f"Altitude: {alt:.2f} m")
        if alt >= target_altitude * 0.95:
            print("Reached target altitude")
            break
        time.sleep(1)

def fly_mission(waypoints):
    for wp in waypoints:

Step 2: Multispectral Image Preprocessing (Python + OpenCV / NumPy)
Goal: Correct raw multispectral images, align bands, and calculate vegetation indices like NDVI.
Example NDVI calculation assuming NIR and Red bands as NumPy arrays:
python
CopyEdit
import numpy as np
import cv2

def calculate_ndvi(nir_band, red_band):
    # Convert to float for division
    nir = nir_band.astype(float)
    red = red_band.astype(float)
    # NDVI = (NIR - Red) / (NIR + Red)
    bottom = (nir + red)
    bottom[bottom == 0] = 0.01  # Avoid division by zero
    ndvi = (nir - red) / bottom
    # Scale NDVI to 0-255 for visualization
    ndvi_scaled = ((ndvi + 1) / 2 * 255).astype(np.uint8)
    return ndvi_scaled

# Load bands (example filenames)
nir_band = cv2.imread('nir_band.tif', cv2.IMREAD_GRAYSCALE)
red_band = cv2.imread('red_band.tif', cv2.IMREAD_GRAYSCALE)

ndvi_image = calculate_ndvi(nir_band, red_band)
cv2.imwrite('ndvi_output.png', ndvi_image)

Step 5: Real-Time Inference on Edge Device (Python + TensorFlow Lite)
Example loading TFLite model and running inference on drone images:
python
CopyEdit
import numpy as np
import cv2
import tflite_runtime.interpreter as tflite

# Load TFLite model and allocate tensors
interpreter = tflite.Interpreter(model_path='model_quantized.tflite')
interpreter.allocate_tensors()

input_details = interpreter.get_input_details()
output_details = interpreter.get_output_details()

def preprocess_image(image_path):
    img = cv2.imread(image_path)
    img = cv2.resize(img, (64, 64))
    img = img.astype(np.float32) / 255.0
    img = np.expand_dims(img, axis=0)  # Add batch dimension
    return img

def predict(image):
    interpreter.set_tensor(input_details[0]['index'], image)
    interpreter.invoke()
    output_data = interpreter.get_tensor(output_details[0]['index'])
    return np.argmax(output_data)

# Example usage:
image = preprocess_image('captured_image.png')
result = predict(image)
print("Predicted class:", result)

import cv2
import numpy as np

def compute_ndvi(nir, red):
    nir = nir.astype(float)
    red = red.astype(float)
    bottom = (nir + red)
    bottom[bottom == 0] = 0.01  # avoid division by zero
    ndvi = (nir - red) / bottom
    return np.clip(ndvi, -1, 1)

nir_img = cv2.imread("nir.jpg", cv2.IMREAD_GRAYSCALE)
red_img = cv2.imread("red.jpg", cv2.IMREAD_GRAYSCALE)

CopyEdit
import onnxruntime as ort
import numpy as np
import cv2

session = ort.InferenceSession("model.onnx")

def predict(image_path):
    img = cv2.imread(image_path)
    img = cv2.resize(img, (128, 128))
    img = img.astype(np.float32) / 255.0
    img = np.expand_dims(img, axis=0)
    inputs = {session.get_inputs()[0].name: img}
    outputs = session.run(None, inputs)
    return np.argmax(outputs[0])

result = predict("ndvi_map.jpg")
print("Prediction:", result)

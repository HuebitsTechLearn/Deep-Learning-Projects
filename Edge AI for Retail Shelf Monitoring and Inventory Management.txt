import torch
import cv2
import pandas as pd

# Load pretrained or fine-tuned YOLOv5 model
model = torch.hub.load('ultralytics/yolov5', 'custom', path='best.pt')  # Use your trained model here

# Initialize camera (0 = webcam, or provide RTSP link or video file)
cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    if not ret:
        break

    # Inference
    results = model(frame)

    # Process results
    df = results.pandas().xyxy[0]  # Bounding box dataframe
    for _, row in df.iterrows():
        label = row['name']
        conf = row['confidence']
        x1, y1, x2, y2 = map(int, [row['xmin'], row['ymin'], row['xmax'], row['ymax']])
        cv2.rectangle(frame, (x1, y1), (x2, y2), (0,255,0), 2)
        cv2.putText(frame, f"{label} {conf:.2f}", (x1, y1 - 10),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0,255,255), 2)

    cv2.imshow('Retail Shelf Monitor', frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()

from flask import Flask, request, send_file, jsonify
import os
import hashlib

app = Flask(__name__)

FIRMWARE_PATH = 'firmware/latest_firmware.bin'
MODEL_PATH = 'models/latest_model.pt'

def file_hash(filepath):
    h = hashlib.sha256()
    with open(filepath, 'rb') as f:
        for chunk in iter(lambda: f.read(4096), b''):
            h.update(chunk)
    return h.hexdigest()

@app.route('/firmware', methods=['GET'])
def get_firmware():
    client_hash = request.headers.get('X-Client-Firmware-Hash')
    server_hash = file_hash(FIRMWARE_PATH)
    if client_hash == server_hash:
        return jsonify({'update': False, 'message': 'Firmware up-to-date'}), 200
    else:
        return send_file(FIRMWARE_PATH, mimetype='application/octet-stream')

@app.route('/model', methods=['GET'])
def get_model():
    client_hash = request.headers.get('X-Client-Model-Hash')
    server_hash = file_hash(MODEL_PATH)
    if client_hash == server_hash:
        return jsonify({'update': False, 'message': 'Model up-to-date'}), 200
    else:
        return send_file(MODEL_PATH, mimetype='application/octet-stream')

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)

import requests
import hashlib
import os
import time
import shutil

SERVER_URL = 'http://your-ota-server-ip:5000'
FIRMWARE_PATH = '/path/to/firmware.bin'
MODEL_PATH = '/path/to/model.pt'

def file_hash(filepath):
    if not os.path.exists(filepath):
        return None
    h = hashlib.sha256()
    with open(filepath, 'rb') as f:
        for chunk in iter(lambda: f.read(4096), b''):
            h.update(chunk)
    return h.hexdigest()

def download_file(url, dest):
    r = requests.get(url, stream=True)
    if r.status_code == 200:
        with open(dest + '.tmp', 'wb') as f:
            for chunk in r.iter_content(1024):
                f.write(chunk)
        os.rename(dest + '.tmp', dest)
        return True
    return False

def ota_update():
    # Firmware
    fw_hash = file_hash(FIRMWARE_PATH)
    fw_resp = requests.get(SERVER_URL + '/firmware', headers={'X-Client-Firmware-Hash': fw_hash})
    if fw_resp.status_code == 200 and fw_resp.headers.get('Content-Type') != 'application/json':
        print('Updating firmware...')
        download_file(SERVER_URL + '/firmware', FIRMWARE_PATH)
        # Trigger firmware flashing logic here or reboot device safely
    else:
        print('Firmware up to date.')

    # Model
    model_hash = file_hash(MODEL_PATH)
    model_resp = requests.get(SERVER_URL + '/model', headers={'X-Client-Model-Hash': model_hash})
    if model_resp.status_code == 200 and model_resp.headers.get('Content-Type') != 'application/json':
        print('Updating AI model...')
        download_file(SERVER_URL + '/model', MODEL_PATH)
        # Reload model in inference pipeline as needed
    else:
        print('Model up to date.')

if __name__ == '__main__':
    while True:
        try:
            ota_update()
        except Exception as e:
            print(f'OTA update failed: {e}')
        time.sleep(3600)  # Check every hour

recognizer = sr.Recognizer()

def nlu_process(text):
    # Simple example NLU: parse commands with keywords
    if 'turn on' in text:
        return {'action': 'turn_on', 'device': 'light'}
    elif 'turn off' in text:
        return {'action': 'turn_off', 'device': 'light'}
    else:
        return {'action': 'unknown'}

import paho.mqtt.client as mqtt
import json
import time

MQTT_BROKER = "mqtt.yourbroker.com"
MQTT_PORT = 1883
MQTT_TOPIC_ALERTS = "retail/shelf/alerts"

client = mqtt.Client()
client.connect(MQTT_BROKER, MQTT_PORT, 60)

def publish_alert(alert):
    payload = json.dumps(alert)
    client.publish(MQTT_TOPIC_ALERTS, payload)
    print(f"Published alert: {payload}")

# Example alert usage
low_stock_alert = {
    'sku': '123456',
    'shelf_id': 'A1',
    'type': 'low_stock',
    'quantity': 3,
    'timestamp': time.time()
}

publish_alert(low_stock_alert)
client.loop_start()

5.Planogram Compliance Logic (JSON + Python)
python
CopyEdit
import json

# Sample planogram JSON
planogram = {
    "A1": ["SKU123", "SKU124", "SKU125"],
    "A2": ["SKU200", "SKU201"]
}

# Detected products on shelf
detected = {
    "A1": ["SKU123", "SKU125", "SKU126"],  # SKU126 misplaced, SKU124 missing
    "A2": ["SKU200", "SKU201"]
}

def check_planogram(planogram, detected):
    alerts = []
    for shelf, expected_skus in planogram.items():
        detected_skus = detected.get(shelf, [])
        missing = set(expected_skus) - set(detected_skus)
        misplaced = set(detected_skus) - set(expected_skus)
        if missing:
            alerts.append({'shelf': shelf, 'issue': 'missing_items', 'items': list(missing)})
        if misplaced:
            alerts.append({'shelf': shelf, 'issue': 'misplaced_items', 'items': list(misplaced)})
    return alerts

alerts = check_planogram(planogram, detected)
print(alerts)

React frontend snippet (subscribe to WebSocket alerts)
javascript
CopyEdit
import React, { useEffect, useState } from 'react';

function AlertDashboard() {
  const [alerts, setAlerts] = useState([]);

  useEffect(() => {
    const ws = new WebSocket('ws://localhost:5000/ws');
    ws.onmessage = (event) => {
      const alert = JSON.parse(event.data);
      setAlerts(prev => [alert, ...prev]);
    };
    return () => ws.close();
  }, []);

  return (
    <div>
      <h1>Retail Shelf Alerts</h1>
      <ul>
        {alerts.map((alert, i) => (
          <li key={i}>{alert.shelf} - {alert.issue} - {alert.items?.join(', ') || alert.quantity}</li>
        ))}
      </ul>
    </div>
  );
}

export default AlertDashboard;
